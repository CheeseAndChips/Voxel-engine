#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec2 aTex;
layout (location = 2) int byte aId;
out vec2 tex;
out byte id;
uniform mat4 view;
uniform mat4 projection;
void main()
{
	gl_Position = projection * view * vec4(pos.x, pos.y, pos.z, 1.0);
	tex = aTex;
	id = aId;
}
